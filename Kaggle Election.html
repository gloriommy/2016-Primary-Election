In [7]:
caret_reg = function(x, y, method, grid, ...){
  set.seed(1)
  control = trainControl(method="repeatedcv", repeats=1, number=10, verboseIter=TRUE)
  train(x=x, y=y, method=method, tuneGrid=grid, 
        trControl=control, metric="RMSE",
        ...)
}
